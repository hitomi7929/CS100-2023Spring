// stdarg
#include<stdarg.h>

int sum_list(int n, ...);

int main(void) {
	int tmp;

	tmp = sum_list(2, 10, 1);
	printf("tmp = %d...\n", tmp);
	
	tmp = sum_list(2, 13, 10);
	printf("tmp = %d...\n", tmp);

	tmp = sum_list(2);
	printf("tmp = %d...\n", tmp);

	return 0;
}


int sum_list(int n, ...) {
	int sum = 0;
	
	va_list vap;

	va_start(vap, n);
	for (int i = 0; i < n; i++) {
		int tmp = va_arg(vap, int);
		sum += tmp;
	}
	va_end(vap);

	return sum;
}

void swap(int *, int *);


int main(void) {

	int a = 10;
	int b = 9;

	printf("before swap: a = %d, b = %d ...\n", a, b);
	swap(a, b);
	swap(&a, &b);
	printf("after swap: a = %d, b = %d ...\n", a, b);

	return 0;
}

void swap(int a, int b) {
	int t = a;
	a = b;
	b = a;
}


void swap(int *a, int *b) {
	int tmp = *a;
	*a = *b;
	*b = tmp;
}
